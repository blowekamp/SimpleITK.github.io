303  template< class TPixelType, unsigned int VImageDimension, unsigned int VLength,
304  template<typename, unsigned int> class TVector >
305  static Image CastITKToImage( itk::Image< TVector< TPixelType, VLength >, VImageDimension> *img )
306  {
307  typedef itk::VectorImage< TPixelType, VImageDimension > VectorImageType;
308 
309  size_t numberOfElements = img->GetBufferedRegion().GetNumberOfPixels();
310  typename VectorImageType::InternalPixelType* buffer = reinterpret_cast<typename VectorImageType::InternalPixelType*>( img->GetPixelContainer()->GetBufferPointer() );
311 
312  // Unlike an image of Vectors a VectorImage's container is a
313  // container of TPixelType, whos size is the image's number of
314  // pixels * number of pixels per component
315  numberOfElements *= VImageDimension;
316 
317  typename VectorImageType::Pointer out = VectorImageType::New();
318 
319  // Set the image's pixel container to import the pointer provided.
320  out->GetPixelContainer()->SetImportPointer(buffer, numberOfElements, true );
321  img->GetPixelContainer()->ContainerManageMemoryOff();
322  out->CopyInformation( img );
323  out->SetRegions( img->GetBufferedRegion() );
324 
325  return Image(out.GetPointer());
326  }
